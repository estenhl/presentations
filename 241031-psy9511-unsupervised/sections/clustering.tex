\newcommand{\clusterplot}[1]{
    \begin{tikzpicture}
        \begin{axis}[
            height=6cm,
            width=6cm,
            xmajorticks=false,
            ymajorticks=false,
            xlabel=$x_1$,
            ylabel=$x_2$,
        ]
            \ifnum#1=0
                \addplot[
                    only marks,
                    mark=*,
                    blue,
                    opacity=0.5
                ] table [
                    col sep=comma,
                    x=x,
                    y=y
                ]{data/clusterdata.csv};
            \fi
            \ifnum#1=1
                \addplot[
                    only marks,
                    mark=*,
                    blue,
                    opacity=0.5
                ] table [
                    col sep=comma,
                    x=x,
                    y=y,
                    discard if not={c}{0.0}
                ]{data/clusterdata.csv};

                \addplot[
                    only marks,
                    mark=*,
                    green,
                    opacity=0.5
                ] table [
                    col sep=comma,
                    x=x,
                    y=y,
                    discard if not={c}{1.0}
                ]{data/clusterdata.csv};

                \addplot[
                    only marks,
                    mark=*,
                    red,
                    opacity=0.5
                ] table [
                    col sep=comma,
                    x=x,
                    y=y,
                    discard if not={c}{2.0}
                ]{data/clusterdata.csv};
            \fi
        \end{axis}
    \end{tikzpicture}
}
