\newcommand{\predictorplot}[3]{
    \nextgroupplot[
        xlabel=\footnotesize{#1},
        ylabel=\footnotesize{#2},
        ytick=#3
    ]
        \addplot[
            only marks,
            blue,
            opacity=0.5
        ] table [
            col sep=comma,
            x=#1,
            y=mpg
        ] {data/scaledauto.csv};
}

\newcommand{\coefplot}[4]{

        };
}
