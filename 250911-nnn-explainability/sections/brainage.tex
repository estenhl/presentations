\newcommand{\datasettrace}[4]{
    \def\tracesex{#1}
    \def\tracesign{#2}
    \def\currentdataset{#3}
    \def\previousdataset{#4}

    \addplot[
        draw=none,
        line width=0pt,
        name path=trace-#1-#3
    ] table [
        x=age,
        y expr=#2 * \thisrow{#1_#3}
    ]{\data};

    \addplot[
        fill=#3!50
    ] fill between [
        of=trace-#1-#3 and trace-#1-#4
    ];
}

\newcommand{\datasetnode}[4]{
    \node[
        circle,
        anchor=#3,
        draw=#1,
        fill=#1!50,
        label={[text depth=0]right:\scriptsize{#1}},
        inner sep=1.75pt,
        text depth=0
    ] (#4) at #2 {};
}
